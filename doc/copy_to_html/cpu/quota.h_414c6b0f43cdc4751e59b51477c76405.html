     345: /* Quota state flags - they actually come in two flavors - for users and groups */
     346: enum {
     347:     _DQUOT_USAGE_ENABLED = 0,        /* Track disk usage for users */
     348:     _DQUOT_LIMITS_ENABLED,            /* Enforce quota limits for users */
     349:     _DQUOT_SUSPENDED,            /* User diskquotas are off, but
     350:                          * we have necessary info in
     351:                          * memory to turn them on */
     352:     _DQUOT_STATE_FLAGS
     353: };
     354: #define DQUOT_USAGE_ENABLED    (1 << _DQUOT_USAGE_ENABLED)
     355: #define DQUOT_LIMITS_ENABLED    (1 << _DQUOT_LIMITS_ENABLED)
     356: #define DQUOT_SUSPENDED        (1 << _DQUOT_SUSPENDED)
     357: #define DQUOT_STATE_FLAGS    (DQUOT_USAGE_ENABLED | DQUOT_LIMITS_ENABLED | \
     358:                  DQUOT_SUSPENDED)
     359: /* Other quota flags */
     360: #define DQUOT_STATE_LAST    (_DQUOT_STATE_FLAGS * MAXQUOTAS)
     361: #define DQUOT_QUOTA_SYS_FILE    (1 << DQUOT_STATE_LAST)
     362:                         /* Quota file is a special
     363:                          * system file and user cannot
     364:                          * touch it. Filesystem is
     365:                          * responsible for setting
     366:                          * S_NOQUOTA, S_NOATIME flags
     367:                          */
     368: #define DQUOT_NEGATIVE_USAGE    (1 << (DQUOT_STATE_LAST + 1))
     369:                            /* Allow negative quota usage */
     370: 
     371: static inline unsigned int dquot_state_flag(unsigned int flags, int type)
     372: {
     373:     return flags << _DQUOT_STATE_FLAGS * type;
     374: }
     375: 
     376: static inline unsigned int dquot_generic_flag(unsigned int flags, int type)
     377: {
     378:     return (flags >> _DQUOT_STATE_FLAGS * type) & DQUOT_STATE_FLAGS;
     379: }
