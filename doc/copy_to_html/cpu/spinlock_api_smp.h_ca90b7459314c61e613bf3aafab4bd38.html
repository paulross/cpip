      86: static inline int __raw_spin_trylock(raw_spinlock_t *lock)
      87: {
      88:     preempt_disable();
      89:     if (do_raw_spin_trylock(lock)) {
      90:         spin_acquire(&lock->dep_map, 0, 1, _RET_IP_);
      91:         return 1;
      92:     }
      93:     preempt_enable();
      94:     return 0;
      95: }
